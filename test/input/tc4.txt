program OddvsEven;

var
  i, limit, level: Integer;
  choices: Char;

begin
  limit := 5;
  
  for i := 1 to limit do
  begin
    if (i mod 2 == 0) then
      writeln('Genap')
    else
      writeln('Ganjil');
  end;

  while i < 10 do 
  begin
    i := i + 1;
  end;

  repeat
    limit := limit - 1;
  until limit == 0;

  case choices of
    'A': level := 100;
    'B': level := 50;
    'C': level := 10
  end;

  ;
end.
